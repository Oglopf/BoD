\chapter{Analysis}
\underline{Assume}: $\{ {z}_{n} \}$ is a complex valued sequence with $n,m, N \in \mathbb{N}$.
\section{Cauchy Criterion}
\begin{defn}[Cauchy Sequence]
	If $\epsilon > 0$ then $\exists \; N$ such that if $m, n \geq N$ then:
	\[|\; {z}_{n} - {z}_{m} \; | < \epsilon\]
\end{defn}
